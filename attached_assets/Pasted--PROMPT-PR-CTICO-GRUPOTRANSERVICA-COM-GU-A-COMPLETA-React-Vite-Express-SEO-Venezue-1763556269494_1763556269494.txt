async function generateBlogContent(
  template: BlogTemplate,
  city: string,
  sector: string
): Promise<{
  title: string;
  slug: string;
  content: string;
  metaDescription: string;
  keywords: string[];
}> {
  const client = new OpenAI({
    apiKey: process.env.OPENAI_API_KEY,
  });

  // Seleccionar aleatoriamente dentro del template
  const titleTemplate =
    template.titleTemplates[
      Math.floor(Math.random() * template.titleTemplates.length)
    ];
  const slugPattern =
    template.slugPatterns[
      Math.floor(Math.random() * template.slugPatterns.length)
    ];

  // Reemplazar variables
  const title = titleTemplate
    .replace(/{sector}/g, sector)
    .replace(/{city}/g, city)
    .replace(/{year}/g, new Date().getFullYear().toString());

  const slug = slugPattern
    .replace(/{sector}/g, sector.toLowerCase().replace(/\s+/g, '-'))
    .replace(/{city}/g, city.toLowerCase().replace(/\s+/g, '-'));

  const keywords = template.keywords.map((kw) =>
    kw.replace(/{sector}/g, sector).replace(/{city}/g, city)
  );

  // Prompt para OpenAI
  const prompt = `
Escribe un art√≠culo de blog profesional para una empresa de transporte de carga pesada.

T√çTULO: ${title}
PALABRAS CLAVE: ${keywords.join(', ')}

REQUERIMIENTOS:
1. Extensi√≥n: 800-1200 palabras
2. Optimizado para SEO (incluir keywords naturalmente)
3. Estructura: H2 y H3 claros
4. Tono: Profesional, B2B
5. Llamada a acci√≥n al final
6. HTML simple (solo <p>, <h2>, <h3>, <ul>, <li>)

OUTLINE:
${template.outline.map((item) => `- ${item}`).join('\n')}

Enfoque en:
- Ciudad: ${city}
- Sector: ${sector}
- Servicios: Transporte (100-500 ton), Log√≠stica, Almacenamiento

COMIENZA:
`;

  const response = await client.chat.completions.create({
    model: 'gpt-4',
    messages: [{ role: 'user', content: prompt }],
    temperature: 0.7,
    max_tokens: 2000,
  });

  const content = response.choices[0]?.message.content || '';
  const metaDescription = content.substring(0, 155).replace(/<[^>]*>/g, '');

  return {
    title,
    slug,
    content,
    metaDescription,
    keywords,
  };
}

async function generate5BlogsDaily(): Promise<void> {
  console.log('ü§ñ Iniciando generaci√≥n de 5 blogs diarios...');

  const blogsTogenerate = [];

  // Seleccionar 5 templates aleatorios

    const template = TEMPLATES[Math.floor(Math.random() * TEMPLATES.length)];
    const city = CITIES[Math.floor(Math.random() * CITIES.length)];
    const sector = SECTORS[Math.floor(Math.random() * SECTORS.length)];

    blogsTogenerate.push({ template, city, sector });
  }

  // Generar cada blog
  for (const { template, city, sector } of blogsTogenerate) {

      const { title, slug, content, metaDescription, keywords } =

        publishedAt: new Date(),

      });

      console.log(`‚úÖ Blog publicado: "${title}"`);
    } catch (error) {

  console.log('‚úÖ Generaci√≥n completada');
