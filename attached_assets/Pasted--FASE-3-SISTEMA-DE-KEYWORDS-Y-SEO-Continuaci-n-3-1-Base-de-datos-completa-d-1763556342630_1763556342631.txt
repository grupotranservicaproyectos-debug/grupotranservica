### **4.1 Express API para blogs**

```typescript
// src/routes/api/blogs.ts

import express from 'express';
import { db } from '../../db';
import { blogs } from '../../db/schema';
import { desc, eq } from 'drizzle-orm';

const router = express.Router();

// GET /api/blogs - Listar todos
router.get('/', async (req, res) => {
  try {
    const allBlogs = await db
      .select()
      .from(blogs)
      .where(eq(blogs.published, true))
      .orderBy(desc(blogs.publishedAt));
    
    res.json(allBlogs);
  } catch (error) {
    res.status(500).json({ error: 'Error fetching blogs' });
  }
});

// GET /api/blogs/:slug - Blog individual
router.get('/:slug', async (req, res) => {
  try {
    const blog = await db
      .select()
      .from(blogs)
      .where(eq(blogs.slug, req.params.slug));
    
    if (!blog.length) {
      return res.status(404).json({ error: 'Blog not found' });
    }
    
    res.json(blog[0]);
  } catch (error) {
    res.status(500).json({ error: 'Error fetching blog' });
  }
});

// GET /api/blogs/city/:city - Por ciudad
router.get('/city/:city', async (req, res) => {
  try {
    const cityBlogs = await db
      .select()
      .from(blogs)
      .where(eq(blogs.city, req.params.city));
    
    res.json(cityBlogs);
  } catch (error) {
    res.status(500).json({ error: 'Error fetching blogs' });
  }
});

// GET /api/blogs/sector/:sector - Por sector
router.get('/sector/:sector', async (req, res) => {
  try {
```

### **4.2 Sitemap dinámico**

```typescript
// src/routes/api/sitemap.ts

import express from 'express';
import { db } from '../../db';
import { blogs } from '../../db/schema';

const router = express.Router();

router.get('/sitemap.xml', async (req, res) => {
  try {
    const allBlogs = await db.select().from(blogs);
    
    const baseUrl = process.env.SITE_URL || 'https://grupotranservica.com';
    
    let xml = '<?xml version="1.0" encoding="UTF-8"?>\n';
    xml += '<urlset xmlns="http://www.sitemaps.org/schemas/sitemap/0.9">\n';
    
    // URLs principales
    const mainPages = [
      { loc: '/', changefreq: 'weekly', priority: 1.0 },
      { loc: '/servicios', changefreq: 'weekly', priority: 0.9 },
      { loc: '/blog', changefreq: 'daily', priority: 0.9 },
      { loc: '/contacto', changefreq: 'monthly', priority: 0.8 },
      { loc: '/cotizar', changefreq: 'monthly', priority: 0.8 },
    ];
    
    mainPages.forEach((page) => {
      xml += '  <url>\n';
      xml += `    <loc>${baseUrl}${page.loc}</loc>\n`;
      xml += `    <changefreq>${page.changefreq}</changefreq>\n`;
      xml += `    <priority>${page.priority}</priority>\n`;
      xml += '  </url>\n';
    });
    
    // URLs de blogs
    allBlogs.forEach((blog) => {
      xml += '  <url>\n';
      xml += `    <loc>${baseUrl}/blog/${blog.slug}</loc>\n`;
      xml += `    <lastmod>${blog.publishedAt?.toISOString()}</lastmod>\n`;
      xml += '    <changefreq>monthly</changefreq>\n';
      xml += '    <priority>0.8</priority>\n';
      xml += '  </url>\n';
    });
    
    xml += '</urlset>';
    
    res.header('Content-Type', 'application/xml');
    res.send(xml);
  } catch (error) {
    res.status(500).json({ error: 'Error generating sitemap' });
  }
});

export default router;
```
